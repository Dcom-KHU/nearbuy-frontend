'use client';

import store from '@/store/store';
import { Provider } from 'react-redux';

export function Providers({ children }: { children: React.ReactNode }) {
  return <Provider store={store}>{children}</Provider>;
}


import { configureStore } from '@reduxjs/toolkit';
import menuToggleSlice from './menuToggle/menuToggleSlice';

const store = configureStore({
  reducer: {
    menuToggle: menuToggleSlice.reducer,
  },
});

export type RootState = ReturnType<typeof store.getState>;
export type AppDispatch = typeof store.dispatch;
export default store;



import { createSlice } from '@reduxjs/toolkit';

const menuToggleSlice = createSlice({
  name: 'menuToggle',
  initialState: { toggle: false },
  reducers: {
    toggleMenu(state) {
      state.toggle = !state.toggle;
    },
  },
});

export const menuToggleActions = menuToggleSlice.actions;
export default menuToggleSlice;
